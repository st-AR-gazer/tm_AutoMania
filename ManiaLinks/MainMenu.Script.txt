/**
 * Parse a campaign name into a struct containing its season (see Const::C_Season_Names) and its year
 *
 *	@param _CampaignName		The name of the campaign to parse
 *
 *	@return									A struct LibCampaignStruct_K_CampaignName
 */
LibCampaignStruct_K_CampaignName LibCampaignStruct_ParseCampaignName(Text _CampaignName) {
	declare LibCampaignStruct_K_CampaignName CampaignName = LibCampaignStruct_K_CampaignName {
		IsValid = False,
		Name = _CampaignName
	};

	declare Text[] NameSplit = LibCampaignStruct_TL::Split(" ", _CampaignName);
	if (NameSplit.count >= 2) {
		CampaignName.Season = NameSplit[0];
		CampaignName.Year = NameSplit[1];
		CampaignName.IsValid = LibCampaignStruct_C_Season_Names.existskey(CampaignName.Season) && LibCampaignStruct_TL::ToInteger(CampaignName.Year) > 0;
		if (CampaignName.IsValid) {
			CampaignName.TranslatedName = LibCampaignStruct_TL::Compose("%1 %2", LibCampaignStruct_C_Season_Names[CampaignName.Season], CampaignName.Year);
		}
	}

	return CampaignName;
}

// Get the value of the text displayed for the medal score (can be a time or points)
Text LibCampaignStruct_GetMedalScoreText(Integer _Time, Integer _Points, Integer _Respawns, Text _MapType, LibCampaignStruct_K_ScoreTextOptions _Options) {
	declare Text MedalScoreText = "";
	// For Stunt maps, we display the points of the players and not their time
	if (LibCampaignStruct_TitleControl::IsStuntMapType(_MapType)) {
		MedalScoreText = LibCampaignStruct_TM_Const::GetScoreText(_Points, LibCampaignStruct_TM_Const::C_ScoreType_Points, _Options);
	// For Platform maps, we display the respawns of the players
	} else if (LibCampaignStruct_TitleControl::IsPlatformMapType(_MapType)) {
		MedalScoreText = LibCampaignStruct_TM_Const::GetScoreText(_Respawns, LibCampaignStruct_TM_Const::C_ScoreType_Respawns, _Options);
	} else { // other map type (time)
		MedalScoreText = LibCampaignStruct_TM_Const::GetScoreText(_Time, LibCampaignStruct_TM_Const::C_ScoreType_Time, _Options);
	}
	return MedalScoreText;
}

Void SetLabelOpacity(CMlLabel _Label, Boolean _IsFocused) {
	AnimMgr.Flush(_Label);
	if (_IsFocused) {
		AnimMgr.Add(_Label, "<anim opacity=\"1.\" />", 250, CAnimManager::EAnimManagerEasing::QuadOut);
	} else {
		AnimMgr.Add(_Label, "<anim opacity=\"0.8\" />", 250, CAnimManager::EAnimManagerEasing::QuadOut);
	}
}

Void Focus(CMlControl _From, CMlControl _To) {
	if (CMGame_Transition::IsInTransition(This)) return;

	if (_From != Null) {
		ComponentNavigation_ComponentNavigation::Unfocus(This, _From);
		if (CMGame_ExpendableButton::IsButton(_From)) {
			CMGame_ExpendableButton::Unfocus(This, (_From as CMlFrame));
		} else if (Trackmania_Button::IsButton(_From)) {
			Trackmania_Button::Unfocus(This, (_From as CMlFrame));
			if (_From.DataAttributeExists("secondarylabel")) {
				SetLabelOpacity(_From.Parent.GetFirstChild(_From.DataAttributeGet("secondarylabel")) as CMlLabel, False);
			}
		}
	}

	if (_To != Null) {
		ComponentNavigation_ComponentNavigation::Focus(This, _To);
		if (CMGame_ExpendableButton::IsButton(_To)) {
			CMGame_ExpendableButton::Focus(This, (_To as CMlFrame));
		} else if (Trackmania_Button::IsButton(_To)) {
			Trackmania_Button::Focus(This, (_To as CMlFrame));
			if (_To.DataAttributeExists("secondarylabel")) {
				SetLabelOpacity(_To.Parent.GetFirstChild(_To.DataAttributeGet("secondarylabel")) as CMlLabel, True);
			}
		}
	}
}

	declare CMlFrame Frame_Controller = (State.Controls.Frame_Global.GetFirstChild("frame-controller") as CMlFrame);

	declare CMlFrame Frame_DifficultyController = (State.Controls.Frame_Global.GetFirstChild("frame-difficulty-gamepad") as CMlFrame);
	declare CMlFrame Frame_Mood = (State.Controls.Frame_Global.GetFirstChild("frame-mood") as CMlFrame);

		ComponentNavigation_ComponentNavigation::GetFocusedControl(This, "navgroup-page-mapeditor-settings").ControlId == State.Controls.Button_Back.ControlId
	);

	NavigationMemory = [];
	foreach (Step => NavGroup in C_SettingsStep_NavGroups) {
		ComponentNavigation_ComponentNavigation::EnableGroup(This, NavGroup, Step == _Step);
	}

	// Frames visibility
	State.Controls.Frame_MainMenu.Visible = _Step == C_SettingsStep_MainMenu;
	Frame_CreateType.Visible = _Step == C_SettingsStep_CreateType;
	Frame_Controller.Visible = _Step == C_SettingsStep_Controller;
	Frame_DifficultyMouse.Visible = _Step == C_SettingsStep_DifficultyMouse;
	Frame_DifficultyController.Visible = _Step == C_SettingsStep_DifficultyGamepad;
	Frame_Mood.Visible = _Step == C_SettingsStep_Mood;
	if (_Step != C_SettingsStep_BrowseMaps) {
		DisplayTrackBrowser(State, False, False);
	}

	// Update step
	State.CurrentStep = _Step;
	switch (State.CurrentStep) {
		case C_SettingsStep_MainMenu: {
			Focus(ComponentNavigation_ComponentNavigation::GetFocusedControl(This, "navgroup-page-mapeditor-settings"), State.Controls.Button_Create);
			ComponentNavigation_ComponentNavigation::ChangeControlTarget(This, State.Controls.Button_Back, "up", State.Controls.Button_Create.ControlId);
			ComponentNavigation_ComponentNavigation::ChangeControlTarget(This, State.Controls.Button_Back, "right", State.Controls.Button_Create.ControlId);
		}

			DisplayTrackBrowser(State, True, True);
		}
		case C_SettingsStep_CreateType: {
			Focus(ComponentNavigation_ComponentNavigation::GetFocusedControl(This, "navgroup-page-mapeditor-settings"), State.Controls.Button_CreateRace);
			ComponentNavigation_ComponentNavigation::ChangeControlTarget(This, State.Controls.Button_Back, "up", State.Controls.Button_CreateStunt.ControlId);
			ComponentNavigation_ComponentNavigation::ChangeControlTarget(This, State.Controls.Button_Back, "right", State.Controls.Button_CreateStunt.ControlId);
		}
		case C_SettingsStep_Controller: {
			if (Platform::IsFromFirstParty(System)) {
				State.LaunchParams.Controller = C_Controller_Gamepad;
				State = UpdateStep(State, C_SettingsStep_DifficultyGamepad);
			} else {
				declare CMlControl Button_Mouse = Frame_Controller.GetFirstChild("button-mouse");
				Focus(ComponentNavigation_ComponentNavigation::GetFocusedControl(This, "navgroup-page-mapeditor-settings"), Button_Mouse);
				ComponentNavigation_ComponentNavigation::ChangeControlTarget(This, State.Controls.Button_Back, "up", Button_Mouse.ControlId);
				ComponentNavigation_ComponentNavigation::ChangeControlTarget(This, State.Controls.Button_Back, "right", Button_Mouse.ControlId);
			}
		}
		case C_SettingsStep_DifficultyMouse: {
			declare CMlControl Button_MouseSimple = Frame_DifficultyMouse.GetFirstChild("button-difficulty-mouse-simple");
			Focus(ComponentNavigation_ComponentNavigation::GetFocusedControl(This, "navgroup-page-mapeditor-settings"), Button_MouseSimple);
			ComponentNavigation_ComponentNavigation::ChangeControlTarget(This, State.Controls.Button_Back, "up", Button_MouseSimple.ControlId);
			ComponentNavigation_ComponentNavigation::ChangeControlTarget(This, State.Controls.Button_Back, "right", Button_MouseSimple.ControlId);
		}
		case C_SettingsStep_DifficultyGamepad: {
			Focus(ComponentNavigation_ComponentNavigation::GetFocusedControl(This, "navgroup-page-mapeditor-settings"), Frame_DifficultyController.GetFirstChild("button-difficulty-gamepad-simple"));
			ComponentNavigation_ComponentNavigation::ChangeControlTarget(This, State.Controls.Button_Back, "up", "button-difficulty-gamepad-advanced");
			ComponentNavigation_ComponentNavigation::ChangeControlTarget(This, State.Controls.Button_Back, "right", "button-difficulty-gamepad-advanced");
		}
		case C_SettingsStep_Mood: {
			Focus(ComponentNavigation_ComponentNavigation::GetFocusedControl(This, "navgroup-page-mapeditor-settings"), Frame_Mood.GetFirstChild("button-mood-48x48Screen155Day"));
			ComponentNavigation_ComponentNavigation::ChangeControlTarget(This, State.Controls.Button_Back, "up", "button-mood-48x48Screen155Sunset");
			ComponentNavigation_ComponentNavigation::ChangeControlTarget(This, State.Controls.Button_Back, "right", "button-mood-48x48Screen155Sunset");
		}
	}

	if (BackButtonFocused) {
		Focus(ComponentNavigation_ComponentNavigation::GetFocusedControl(This, "navgroup-page-mapeditor-settings"), State.Controls.Button_Back);
	}

	return State;
}

K_State Cancel(K_State _State) {
	declare K_State State = _State;

	switch (State.CurrentStep) {
		case C_SettingsStep_MainMenu: Router_Router::PushParent(This);
		case C_SettingsStep_BrowseMaps: State = UpdateStep(State, C_SettingsStep_MainMenu);
		case C_SettingsStep_CreateType: State = UpdateStep(State, C_SettingsStep_MainMenu);
		case C_SettingsStep_Controller: {
			if (
				Router_Router::GetCurrentParentPath(This) == "/create/server-review" ||
				Router_Router::GetCurrentParentPath(This) == "/local/multi/settings"
			) {
				Router_Router::PushParent(This);
			} else if (Router_Router::GetCurrentParentPath(This) == "/clubs/details/create/campaign") {
				declare CampaignStruct::LibCampaignStruct_K_Map Map;
				declare Router_Router::K_URL_Config Config = Router_Router::K_URL_Config {
					SaveHistory = True,
					ResetPreviousPagesDisplayed = True,
					KeepPreviousPagesDisplayed = False,
					HidePreviousPage = True,
					ShowParentPage = True
				};
				Router_Router::Push(This, "/clubs/details/create/campaign", ["Map" => Map.tojson()], Config);
			} else if (Platform::IsFromFirstParty(System)) {
				State = UpdateStep(State, C_SettingsStep_MainMenu);
			} else {
				State = UpdateStep(State, C_SettingsStep_CreateType);
			}
		}
		case C_SettingsStep_DifficultyMouse: {
			if (TitleControl::IsStuntMapType(State.LaunchParams.MapType) || TitleControl::IsPlatformMapType(State.LaunchParams.MapType)) {
				State = UpdateStep(State, C_SettingsStep_CreateType);
			} else {
				State = UpdateStep(State, C_SettingsStep_Controller);
			}
		}
		case C_SettingsStep_DifficultyGamepad: {
			if (Platform::IsFromFirstParty(System)) {
				State.CurrentStep = C_SettingsStep_Controller;
				State = Cancel(State);
			} else {
				State = UpdateStep(State, C_SettingsStep_Controller);
			}
		}
		case C_SettingsStep_Mood: {
			if (TitleControl::IsRoyalMapType(State.LaunchParams.MapType)) {
				State = UpdateStep(State, C_SettingsStep_CreateType);
			} else if (State.LaunchParams.Controller == C_Controller_Mouse) {
				State = UpdateStep(State, C_SettingsStep_DifficultyMouse);
			} else {
				State = UpdateStep(State, C_SettingsStep_DifficultyGamepad);
			}
		}
	}

	return State;
}

K_State LaunchEditor(K_State _State) {
	declare K_State State = _State;
	declare Text[] EditorPluginScripts;
	declare Text[] EditorPluginSettings;

	// When opening the gamepad editor, we don't want to allow plugins that were saved in the profile
	// opened in previous Map Editor session to be opened alongside GamepadTrackEditor.Script.txt
	declare Boolean	OnlyUseForcedPlugins = TitleControl::IsRaceMapType(State.LaunchParams.MapType) && State.LaunchParams.Controller == C_Controller_Gamepad;

	// Add the 'PlayMapAtSave' plugin
	EditorPluginScripts.add("TrackMania/Nadeo/Trackmania/PlayMapAtSave.Script.txt");
	if (
		Router_Router::GetCurrentParentPath(This) == "/clubs/details/create/campaign" ||
		TitleControl::IsStuntMapType(State.LaunchParams.MapType) ||
		TitleControl::IsPlatformMapType(State.LaunchParams.MapType) ||
		TitleControl::IsRoyalMapType(State.LaunchParams.MapType)
	) {
		EditorPluginSettings.add("<root><setting name=\"S_ShowMenuAfterSave\" value=\""^False^"\" type=\"boolean\"/></root>");
	} else {
		EditorPluginSettings.add("");
	}

	// Add the 'GamepadTrackEditor' plugin (only available for race maps)
	if (State.LaunchParams.Controller == C_Controller_Gamepad && TitleControl::IsRaceMapType(State.LaunchParams.MapType)) {
		EditorPluginScripts.add("TrackMania/Nadeo/TMEditor/GamepadTrackEditor.Script.txt");
		EditorPluginSettings.add("<root><setting name=\"S_EditorLevel\" value=\""^State.LaunchParams.Difficulty^"\" type=\"integer\"/></root>");
	}

	// Add the 'ColorAndAnimation' plugin (except in the Gamepad Editor)
	if (State.LaunchParams.Controller != C_Controller_Gamepad) {
		EditorPluginScripts.add("TrackMania/Nadeo/TMGame/ColorAndAnimation.Script.txt");
		EditorPluginSettings.add("");
	}

	// Update the loading screen
	declare Text LoadingScreenTitle = Tools::Ternary(State.LaunchParams.MapToEdit.Name != "", State.LaunchParams.MapToEdit.Name, "Track editor");
	LoadingScreenStore::UpdateTitleMenuLoadingScreen_Random(This, LoadingScreenTitle);

	// Launch the editor
	Share::ResetLastMapCreated();
	if (State.LaunchParams.MapToEdit.Uid != "") {
		declare Text FileName = State.LaunchParams.MapToEdit.LocalFileName;
		if (FileName == "") FileName = State.LaunchParams.MapToEdit.DownloadUrl;
		TitleControl::EditMap(
			TitleControl,
			FileName,
			State.LaunchParams.Mood,
			"",
			EnvironmentInfo::C_PlayerModel_CarSport,
			EditorPluginScripts,
			EditorPluginSettings,
			State.LaunchParams.Difficulty != C_Difficulty_Simple,
			OnlyUseForcedPlugins
		);
	} else if (TitleControl::IsRoyalMapType(State.LaunchParams.MapType)) {
		TitleControl::EditNewMapFromBaseMap(
			TitleControl,
			"RoyalBaseMap.Map.Gbx",
			State.LaunchParams.Mood,
			"",
			EnvironmentInfo::C_PlayerModel_CarSport,
			State.LaunchParams.MapType,
			EditorPluginScripts,
			EditorPluginSettings,
			OnlyUseForcedPlugins
		);
	} else {
		TitleControl::EditNewMap(
			TitleControl,
			EnvironmentInfo::C_Collection_Stadium,
			State.LaunchParams.Mood,
			"",
			EnvironmentInfo::C_PlayerModel_CarSport,
			State.LaunchParams.MapType,
			State.LaunchParams.Controller == C_Controller_Mouse && State.LaunchParams.Difficulty == C_Difficulty_Simple,
			EditorPluginScripts,
			EditorPluginSettings,
			OnlyUseForcedPlugins
		);
	}

	// Coming back from the editor
	declare Share::K_MapInfo MapInfo = Share::GetLastMapCreated();
	DisplayTrackBrowser(State, False, False);
	if (Share::GetGoToServerReview()) {
		if (Trackmania_UserStore::HasGamePermission(PrivilegeMgr, RBAC::C_GamePermission_AccessServerReview)) {
			declare CampaignStruct::LibCampaignStruct_K_Map Map = Share::ConvertToMap(MapInfo);
			Router_Router::SetParentPath(This, "/create/server-review", Router_Router::GetCurrentRoutePath(This));
			Router_Router::Push(This, "/create/server-review", ["Map" => Map.tojson(), "IsWeeklyShortsReview" => ""^Share::GetGoToWeeklyShortsReview()]);
		} else {
			ClubEdition::GoToPage(This, "/create", RBAC::GetGamePermissionMessage(RBAC::C_GamePermissionMessage_AccessServerReview));
		}
	} else if (Share::GetGoToHotSeat()) {
		if (Trackmania_UserStore::HasGamePermission(PrivilegeMgr, RBAC::C_GamePermission_PlayHotSeat)) {
			declare CampaignStruct::LibCampaignStruct_K_Map Map = Share::ConvertToMap(MapInfo);
			Router_Router::Push(This, "/local/multi/settings", ["Map" => Map.tojson(), "LocalMultiType" => "0"]);
		} else {
			ClubEdition::GoToPage(This, "/local", RBAC::GetGamePermissionMessage(RBAC::C_GamePermissionMessage_PlayHotSeat));
		}
	} else if (Router_Router::GetCurrentParentPath(This) == "/clubs/details/create/campaign") {
		declare CampaignStruct::LibCampaignStruct_K_Map Map = Share::ConvertToMap(MapInfo);
		declare Router_Router::K_URL_Config Config = Router_Router::K_URL_Config {
			SaveHistory = True,
			ResetPreviousPagesDisplayed = True,
			KeepPreviousPagesDisplayed = False,
			HidePreviousPage = True,
			ShowParentPage = True
		};
		Router_Router::Push(This, "/clubs/details/create/campaign", ["Map" => Map.tojson()], Config);
	} else {
		if (MapInfo.Uid != "") {
			State = UpdateStep(State, C_SettingsStep_BrowseMaps);
		} else {
			State = UpdateStep(State, C_SettingsStep_MainMenu);
		}
	}
	Share::ResetLastMapCreated();

	return State;
}

Text GetQuickStartEditorMood() {
	if (UserMgr != Null && UserMgr.MainUserProfile != Null && UserMgr.MainUserProfile.Editors_MapEditorUseQuickstart) {
		switch (UserMgr.MainUserProfile.Editors_MapEditorQuickstartMood) {
			case CUserV2Profile::EMapEditorMood::Sunrise: return C_Mood_Sunrise;
			case CUserV2Profile::EMapEditorMood::Day: return C_Mood_Day;
			case CUserV2Profile::EMapEditorMood::Sunset: return C_Mood_Sunset;
			case CUserV2Profile::EMapEditorMood::Night: return C_Mood_Night;
		}
	}
	return C_Mood_Day;
}

Integer GetQuickStartEditorDifficulty() {
	if (UserMgr != Null && UserMgr.MainUserProfile != Null && UserMgr.MainUserProfile.Editors_MapEditorUseQuickstart) {
		switch (UserMgr.MainUserProfile.Editors_MapEditorQuickstartDifficulty) {
			case CUserV2Profile::EMapEditorDifficulty::Simple: return C_Difficulty_Simple;
			case CUserV2Profile::EMapEditorDifficulty::Advanced: return C_Difficulty_Advanced;
			case CUserV2Profile::EMapEditorDifficulty::Expert: return C_Difficulty_Expert;
		}
	}
	return C_Difficulty_Simple;
}

Integer GetQuickStartEditorGamepad() {
	if (UserMgr != Null && UserMgr.MainUserProfile != Null && UserMgr.MainUserProfile.Editors_MapEditorUseQuickstart) {
		if (UserMgr.MainUserProfile.Editors_MapEditorQuickstartUseGamepad || Platform::IsFromFirstParty(System)) {
			return C_Controller_Gamepad;
		}
	}
	return C_Controller_Mouse;
}

Text GetQuickStartEditorMapType() {
	if (UserMgr != Null && UserMgr.MainUserProfile != Null && UserMgr.MainUserProfile.Editors_MapEditorUseQuickstart) {
		if (Platform::IsFromFirstParty(System)) return "TrackMania\\TM_Race"; //< Players can only create/edit Race maps on console
		switch (UserMgr.MainUserProfile.Editors_MapEditorQuickstartMapType) { //< Check the 'C_SettingId_SystemTrackEditorQuickstartMapType' setting for more information
			case 1: return "TrackMania\\TM_Race";
			case 2: return "TrackMania\\TM_Royal";
			case 3: return "TrackMania\\TM_Stunt";
			case 4: return "TrackMania\\TM_Platform";
		}
	}
	return "";
}

K_State Select(K_State _State, CMlControl _SelectedControl) {
	if (_SelectedControl == Null || CMGame_Transition::IsInTransition(This)) return _State;
	declare K_State State = _State;

	switch (_SelectedControl.ControlId) {
		case "button-back": State = Cancel(State);
		case "button-edit": State = UpdateStep(State, C_SettingsStep_BrowseMaps);
		case "button-club-items": {
			Router_Router::SetParentPath(This, "/create/library-clubitems", Router_Router::GetCurrentRoutePath(This));
			Router_Router::Push(This, "/create/library-clubitems");
		}
		case "button-create": {
			if (DataFileMgr.UserSave_AvailableSpace < C_MinSpaceNeededForTrackCreationInkB) {
				OpenDiskSpacePopUp(_State.Controls.Frame_MultiChoice);
			} else {
				State.LaunchParams = K_LaunchParams {
					Controller = GetQuickStartEditorGamepad(),
					Difficulty = GetQuickStartEditorDifficulty(),
					MapType = GetQuickStartEditorMapType(),
					Mood = GetQuickStartEditorMood(),
					MapToEdit = CampaignStruct::LibCampaignStruct_K_Map {}
				};
				if (IsQuickStartEditorEnabled() && State.LaunchParams.MapType != "") {
					State = LaunchEditor(State);
				} else if (Platform::IsFromFirstParty(System)) {
					State = Select(State, State.Controls.Button_CreateRace);
				} else {
					State = UpdateStep(State, C_SettingsStep_CreateType);
				}
			}
		}
		case "button-create-race": {
			State.LaunchParams.MapType = "TrackMania\\TM_Race";
			if (IsQuickStartEditorEnabled()) {
				State = LaunchEditor(State);
			} else {
				State = UpdateStep(State, C_SettingsStep_Controller);
			}
		}
		case "button-create-royal": {
			State.LaunchParams.MapType = "TrackMania\\TM_Royal";
			if (IsQuickStartEditorEnabled()) {
				State = LaunchEditor(State);
			} else if (Trackmania_UserStore::HasGamePermission(PrivilegeMgr, RBAC::C_GamePermission_OpenAdvancedMapEditor)) {
				State = UpdateStep(State, C_SettingsStep_Mood); //< Simple editor not available when creating from base map
			} else {
				ClubEdition::GoToPage(This, Router_Router::GetCurrentRoutePath(This), RBAC::GetGamePermissionMessage(RBAC::C_GamePermissionMessage_OpenAdvancedMapEditor));
			}
		}
		case "button-create-stunt": {
			State.LaunchParams.MapType = "TrackMania\\TM_Stunt";
			if (IsQuickStartEditorEnabled()) {
				State = LaunchEditor(State);
			} else {
				State = UpdateStep(State, C_SettingsStep_DifficultyMouse);
			}
		}
		case "button-create-platform": {
			State.LaunchParams.MapType = "TrackMania\\TM_Platform";
			if (IsQuickStartEditorEnabled()) {
				State = LaunchEditor(State);
			} else {
				State = UpdateStep(State, C_SettingsStep_DifficultyMouse);
			}
		}
		case "button-mouse": {
			State.LaunchParams.Controller = C_Controller_Mouse;
			// Editing a map done in advanced editor
			if (State.LaunchParams.MapToEdit.Uid != "" && !State.LaunchParams.MapToEdit.CreatedWithSimpleEditor) {
				if (!Trackmania_UserStore::HasGamePermission(PrivilegeMgr, RBAC::C_GamePermission_OpenAdvancedMapEditor)) {
					ClubEdition::GoToPage(This, Router_Router::GetCurrentRoutePath(This), RBAC::GetGamePermissionMessage(RBAC::C_GamePermissionMessage_OpenAdvancedMapEditor));
				} else {
					State = UpdateStep(State, C_SettingsStep_Mood);
				}
			} else {
				State = UpdateStep(State, C_SettingsStep_DifficultyMouse);
			}
		}
		case "button-gamepad": {
			State.LaunchParams.Controller = C_Controller_Gamepad;
			State = UpdateStep(State, C_SettingsStep_DifficultyGamepad);
		}
		default: {
			if (_SelectedControl.DataAttributeExists("difficulty")) {
				State.LaunchParams.Difficulty = TL::ToInteger(_SelectedControl.DataAttributeGet("difficulty"));
				if (State.LaunchParams.Difficulty == C_Difficulty_Simple && Trackmania_UserStore::HasGamePermission(PrivilegeMgr, RBAC::C_GamePermission_OpenSimpleMapEditor)) {
					State = LaunchEditor(State);
				} else if (State.LaunchParams.Difficulty != C_Difficulty_Simple && Trackmania_UserStore::HasGamePermission(PrivilegeMgr, RBAC::C_GamePermission_OpenAdvancedMapEditor)) {
					State = UpdateStep(State, C_SettingsStep_Mood);
				} else {
					if (State.LaunchParams.Difficulty == C_Difficulty_Simple) {
						ClubEdition::GoToPage(This, Router_Router::GetCurrentRoutePath(This), RBAC::GetGamePermissionMessage(RBAC::C_GamePermissionMessage_OpenSimpleMapEditor));
					} else {
						ClubEdition::GoToPage(This, Router_Router::GetCurrentRoutePath(This), RBAC::GetGamePermissionMessage(RBAC::C_GamePermissionMessage_OpenAdvancedMapEditor));
					}
				}
			} else if (_SelectedControl.DataAttributeExists("mood")) {
				State.LaunchParams.Mood = _SelectedControl.DataAttributeGet("mood");
				State = LaunchEditor(State);
			}
		}
	}

	return State;
}

			State.Controls.Label_QuickStartCreate.Visible = (UserMgr.MainUserProfile.Editors_MapEditorUseQuickstart && (Platform::IsFromFirstParty(System) || UserMgr.MainUserProfile.Editors_MapEditorQuickstartMapType > 0));
		}
		if (State.Controls.Label_QuickStartMapType.Visible != UserMgr.MainUserProfile.Editors_MapEditorUseQuickstart) {
			State.Controls.Label_QuickStartMapType.Visible = UserMgr.MainUserProfile.Editors_MapEditorUseQuickstart;
		}
	}

	if (State.CurrentStep != C_SettingsStep_BrowseMaps) {
		foreach (Event in ComponentNavigation_ComponentNavigation::GetPendingEvents(This)) {
			switch (Event.Type) {
				case ComponentNavigation_ComponentNavigation::C_EventType_NavigateInput: {
					if (ComponentNavigation_ComponentNavigation::IsEventFromGroup(This, Event, "navgroup-page-mapeditor-settings")) {
						switch (Event.Input) {
							case CMlScriptEvent::EMenuNavAction::Select: State = Select(State, Event.To);
							case CMlScriptEvent::EMenuNavAction::Cancel: State = Cancel(State);
							default: {
								declare CMlControl To = Event.To;
								if (Event.To != Null && ComponentNavigation_ComponentNavigation::GetControlTarget(Event.From, Event.Input) == Event.To) {
									declare CMlControl[CMlScriptEvent::EMenuNavAction] NavigationMemory for Event.To;
									NavigationMemory[ComponentNavigation_ComponentNavigation::InvertInput(Event.Input)] = Event.From;
								}
								if (Event.From != Null) {
									declare CMlControl[CMlScriptEvent::EMenuNavAction] NavigationMemory for Event.From;
									if (NavigationMemory.existskey(Event.Input) && !ComponentNavigation_ComponentNavigation::IsLocked(NavigationMemory.get(Event.Input, Null))) {
										To = NavigationMemory[Event.Input];
									}
								}
								Focus(Event.From, To);
							}
						}
					}
				}
				case ComponentNavigation_ComponentNavigation::C_EventType_NavigateMouse: {
					if (ComponentNavigation_ComponentNavigation::IsEventFromGroup(This, Event, "navgroup-page-mapeditor-settings")) {
						switch (Event.Mouse) {
							case CMlScriptEvent::Type::MouseClick: State = Select(State, Event.To);
							case CMlScriptEvent::Type::MouseOver: Focus(Event.From, Event.To);
						}
					}
				}
			}
		}
	} else {
		foreach (Event in Trackmania_TrackBrowser::GetPendingEvents(This)) {
			switch (Event.Type) {
				case Trackmania_TrackBrowser::C_Event_Select: {
					State.LaunchParams.MapToEdit = Event.Track;
					State.LaunchParams.MapType = State.LaunchParams.MapToEdit.Type;
					if (Platform::IsFromFirstParty(System) && !State.LaunchParams.MapToEdit.CreatedWithGamepadEditor) {
						OpenGamepadEditorPopUp(State.Controls.Frame_MultiChoice);
					} else {
						State.LaunchParams.Controller = C_Controller_Gamepad;
						if (Platform::IsPC(System) && !State.LaunchParams.MapToEdit.CreatedWithGamepadEditor) {
							State.LaunchParams.Controller = C_Controller_Mouse;
						}
						State.LaunchParams.Difficulty = C_Difficulty_Expert;
						if (State.LaunchParams.MapToEdit.CreatedWithSimpleEditor) {
							State.LaunchParams.Difficulty = C_Difficulty_Simple;
						}
						State.LaunchParams.Mood = "";
						State = LaunchEditor(State);
					}
				}
				case Trackmania_TrackBrowser::C_Event_Exit: {
					ControllerInfo::SetButtons(C_Buttons_Default);
					State = Cancel(State);
				}
			}
		}
	}

	/* Pop up multi choice events */
	foreach (Event in Trackmania_PopUpMultiChoice::GetPendingEvents(This)) {
		if (Event.PopUpMultiChoice != Null && Event.PopUpMultiChoice.Id == State.Controls.Frame_MultiChoice.Id) {
			switch (Event.Type) {
				case Trackmania_PopUpMultiChoice::C_EventType_Display: {
					Trackmania_TrackBrowser::LockBrowser(This, State.Controls.Frame_MultiChoice.Visible);
					if (C_SettingsStep_NavGroups.existskey(State.CurrentStep)) {
						ComponentNavigation_ComponentNavigation::EnableGroup(This, C_SettingsStep_NavGroups[State.CurrentStep], !State.Controls.Frame_MultiChoice.Visible);
					}
					ComponentNavigation_ComponentNavigation::EnableGroup(This, "navgroup-page-mapeditor-settings", !State.Controls.Frame_MultiChoice.Visible);
				}
			}
		}
	}
}
foreach (Event in Router_Router::GetPendingEvents(This)) {
	switch (Event.Type) {
		case Router_Router::C_Event_LeavingRoute: {
			State = UpdateStep(State, C_SettingsStep_MainMenu);
		}
	}
}
